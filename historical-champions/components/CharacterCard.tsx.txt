import React from 'react';
import { CharacterData } from '../types';
import { Star } from 'lucide-react';

interface Props {
  character: CharacterData;
  isUnlocked: boolean;
  isSelected: boolean;
  onClick: () => void;
}

export const CharacterCard: React.FC<Props> = ({ character, isUnlocked, isSelected, onClick }) => {
  return (
    <div 
      onClick={onClick}
      className={`
        relative aspect-[3/4] rounded-lg border-2 cursor-pointer transition-all duration-200 overflow-hidden group
        ${isSelected ? 'border-yellow-400 scale-105 shadow-[0_0_20px_rgba(250,204,21,0.5)]' : 'border-slate-600 hover:border-slate-400'}
      `}
    >
      {/* Character Image - Completely transparent background */}
      <div className="absolute inset-0 p-2 flex items-end justify-center">
        <img 
            src={character.portraitSrc} 
            alt={character.name} 
            className="w-full h-full object-contain drop-shadow-lg group-hover:scale-110 transition-transform duration-300" 
        />
      </div>

      {/* Info Footer - Gradient only at bottom for text readability */}
      <div className="absolute bottom-0 left-0 right-0 bg-gradient-to-t from-black via-black/80 to-transparent p-2 pt-6 z-10">
        <div className="flex justify-center mb-1">
          {Array.from({ length: character.rarity }).map((_, i) => (
            <Star key={i} size={10} className="text-yellow-400 fill-yellow-400" />
          ))}
        </div>
        <p className="text-white text-xs font-bold text-center truncate">{character.name}</p>
      </div>
    </div>
  );
};
